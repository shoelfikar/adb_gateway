    ```go
    type DisconnectCallback func()

    type ADBWatchdog struct {
        client     *Client
        interval   time.Duration          // probe interval (default 2s)
        onDisconnect DisconnectCallback   // called when ADB becomes unreachable
    }
    ```

    **cmd/gateway/main.go -- Add ADB lifecycle goroutine:**

    Refactor main.go to add an ADB lifecycle loop that runs as a goroutine. The current startup code becomes the "first iteration" of this loop. The key structure:

    ```go
    // After initial startup (AwaitADBReady + Reconcile + WatchDevices):

    // Watchdog probes ADB every 2 seconds. On disconnect, it calls the callback.
    watchdog := adb.NewADBWatchdog(adbClient, 2*time.Second, func() {
        // This callback is invoked when ADB disconnect is detected.
        // Signal the lifecycle loop to enter reconnect mode.
    })

    // The lifecycle loop:
    // 1. Watch for ADB disconnect (via watchdog probe failure OR device watcher channel close)
    // 2. On disconnect:
    //    a. Log "ADB connection lost"
    //    b. registry.MarkAllDisconnected() -- clear stale state
    //    c. Stop the old watchdog
    //    d. reconnector.AwaitADBReady(ctx) -- reconnect with backoff
    //    e. hostServices.ReinitializeGoadb() -- fresh goadb connection
    //    f. reconciler.Reconcile(ctx) -- clean up orphan state
    //    g. Get active session specs, reconnector.ReissueReverseForwards for each
    //    h. Create new device watcher, restart WatchDevices goroutine
    //    i. Restart watchdog
    // 3. Repeat until context cancelled (graceful shutdown)
    ```

    Implementation approach -- use a channel to signal ADB disconnect from the watchdog and from WatchDevices:

    ```go
    adbDisconnected := make(chan struct{}, 1)

    // Watchdog goroutine: probe ADB every 2s, signal on disconnect
    go func() {
        ticker := time.NewTicker(2 * time.Second)
        defer ticker.Stop()
        for {
            select {
            case <-ctx.Done():
                return
            case <-ticker.C:
                probeCtx, cancel := context.WithTimeout(ctx, 3*time.Second)
                err := watchdog.ProbeOnce(probeCtx)
                cancel()
                if err != nil {
                    select {
                    case adbDisconnected <- struct{}{}:
                    default: // already signaled, don't block
                    }
                    return // stop probing; lifecycle loop will restart us
                }
            }
        }
    }()
    ```

    WatchDevices already signals disconnect by returning `true`. Wire it into the same channel:

    ```go
    go func() {
        if registry.WatchDevices(ctx, deviceEvents) {
            // Channel closed = ADB disconnect
            select {
            case adbDisconnected <- struct{}{}:
            default:
            }
        }
        // false return = context cancelled (graceful shutdown)
    }()
    ```

    The reconnect loop (in main goroutine or a dedicated goroutine):

    ```go
    for {
        select {
        case <-ctx.Done():
            return // graceful shutdown
        case <-adbDisconnected:
            slog.Warn("ADB connection lost, starting reconnection")
            registry.MarkAllDisconnected()

            // Reconnect with backoff
            if err := reconnector.AwaitADBReady(ctx); err != nil {
                slog.Error("ADB reconnection cancelled", "error", err)
                return
            }
            slog.Info("ADB server reconnected")

            // Reinitialize goadb so device watcher and host services work
            if err := hostServices.ReinitializeGoadb(); err != nil {
                slog.Error("failed to reinitialize goadb after reconnect", "error", err)
                // Non-fatal: continue, ListDevices/WatchDevices will retry on next cycle
            }

            // Reconcile: clean up orphan processes and stale forwards
            if err := reconciler.Reconcile(ctx); err != nil {
                slog.Warn("post-reconnect reconciliation failed", "error", err)
            }

            // Re-issue reverse forwards for active sessions
            specs := registry.ActiveSessionSpecs()
            for _, entry := range registry.List() {
                if entry.GetState() == session.StateFailed && entry.GetSession() != nil {
                    sess := entry.GetSession()
                    // Build spec for this session's reverse forward
                    deviceSpec := fmt.Sprintf("localabstract:scrcpy_%s", sess.SCID())
                    videoLn := sess.VideoLn()
                    var hostSpec string
                    if videoLn != nil {
                        hostSpec = fmt.Sprintf("tcp:%d", videoLn.Addr().(*net.TCPAddr).Port)
                    } else {
                        slog.Warn("skipping reverse forward re-issuance: video listener is nil",
                            "device", entry.Serial, "session", sess.ID)
                        continue
                    }
                    newMappings, err := reconnector.ReissueReverseForwards(ctx, entry.Serial,
                        []adb.ReverseMappingSpec{{DeviceSpec: deviceSpec, HostSpec: hostSpec}})
                    if err != nil {
                        slog.Error("failed to re-issue reverse forwards",
                            "device", entry.Serial, "error", err)
                        continue
                    }
                    // Update the session's reverse mapping with the new one
                    if len(newMappings) > 0 {
                        entry.Lock()
                        oldRM := entry.Session.ReverseMap()
                        entry.Session.SetReverseMap(newMappings[0])
                        entry.Unlock()
                        if oldRM != nil {
                            oldRM.Close()
                        }
                        slog.Info("re-issued reverse forward",
                            "device", entry.Serial, "session", sess.ID,
                            "device_spec", deviceSpec, "host_spec", hostSpec)
                    }
                }
            }

            // Start new device watcher
            newEvents, err := hostServices.NewDeviceWatcher(ctx)
            if err != nil {
                slog.Error("failed to restart device watcher after reconnect", "error", err)
                // Trigger another reconnect cycle
                select {
                case adbDisconnected <- struct{}{}:
                default:
                }
                continue
            }

            // Restart WatchDevices and watchdog goroutines
            go func() {
                if registry.WatchDevices(ctx, newEvents) {
                    select {
                    case adbDisconnected <- struct{}{}:
                    default:
                    }
                }
            }()
            // Restart watchdog...
            slog.Info("ADB lifecycle restored, device watcher restarted")
        }
    }
    ```

    Note: The reconnection loop needs `DeviceSession` to expose `SetReverseMap(*adb.ReverseMapping)` and `ReverseMap() *adb.ReverseMapping` methods (it already has a private `reverseMap` field). Add these accessor methods to `internal/session/supervisor.go`.
